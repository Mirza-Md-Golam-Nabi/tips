```c
#include<stdio.h>
int main(){
    double pi = 3.14159, r, volume;
    scanf("%lf", &r);
    volume = 4.0/3 * pi * r * r * r;
    printf("VOLUME = %.3lf\n", volume);


    return 0;
}
```

#include<stdio.h>
#include<math.h>
int main(){
